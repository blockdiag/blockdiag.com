Sample diagams¶

simple diagram¶

diagram {
   A -> B -> C -> D;
   A -> E -> F -> G;
}

label attribute to nodes and edges¶

diagram {
   // Set labels to nodes.
   A [label = "foo"];
   B [label = "bar"];
   C [label = "baz"];

   // Set labels to edges. (short text only)
   A -> B [label = "click bar"];
   B -> C [label = "click baz"];
   C -> A;
}

style attribute to nodes and edges¶

diagram {
   // Set boder-style and backgroun-color to nodes.
   A [style = dotted];
   B [style = dashed];
   C [color = pink];
   D [color = "#888888"];

   // Set border-style and color to edges.
   A -> B [style = dotted];
   B -> C [style = dashed];
   C -> D [color = "red"];

   // Set numbered-badge to nodes.
   E [numbered = 99];

   // Set background image to nodes (and erase label).
   F [label = "", background = "source/diagrams/python-logo.gif"];

   // Set arrow direction to edges.
   E -> F [dir = none];
   F -> G [dir = forward];
   G -> H [dir = back];
   H -> I [dir = both];
}

direction of edges¶

New in version 0.5.5.

{
  A -> B <- C -- D <-> E;
}

edge folding¶

New in version 0.6.1.

{
  A -> B -> C -> D -> E;

  // fold edge at C to D (D will be layouted at top level; left side)
  C -> D [folded];
}

mutlilingualization¶

diagram admin {
   // Set M17N text using label property.
   A [label = "起"];
   B [label = "承"];
   C [label = "転"];
   D [label = "結"];

   A -> B -> C -> D;

   // Use M17N text directly (need to quote).
   "春" -> "夏" -> "秋" -> "冬";
}

Note

blockdiag supports UTF-8 encoding only.

shape of nodes¶

New in version 0.6.5: (experimental)

New in version 0.6.6: actor shape

{
  // standard node shapes
  box [shape = "box"];
  roundedbox [shape = "roundedbox"];
  diamond [shape = "diamond"];
  ellipse [shape = "ellipse"];
  note [shape = "note"];

  cloud [shape = "cloud"];
  mail [shape = "mail"];
  beginpoint [shape = "beginpoint"];
  endpoint [shape = "endpoint"];
  minidiamond [shape = "minidiamond"];
  actor [shape = "actor"];

  box -> roundedbox -> diamond -> ellipse;
  cloud -> note -> mail -> actor;
  minidiamond -> beginpoint -> endpoint;


  // node shapes for flowcharts
  condition [shape = "flowchart.condition"];
  database [shape = "flowchart.database"];
  input [shape = "flowchart.input"];
  loopin [shape = "flowchart.loopin"];
  loopout [shape = "flowchart.loopout"];
  terminator [shape = "flowchart.terminator"];

  condition -> database -> terminator -> input;
  loopin -> loopout;
}

grouping nodes¶

New in version 0.5.3: label attributes for group

New in version 0.6: nested groups

diagram admin {
   A -> B -> C -> D;
   A -> E;
   A -> H;

   // A and B belong to first group.
   group {
      A; B;
   }

   // E, F and G belong to second group.
   group {
      // Set group-label
      label = "second group";

      // Set background-color to this group.
      color = "#77FF77";

      E -> F -> G;

      // H and I belong to third "nested" group.
      group {
          color = "#7777FF";
          H -> I;
      }
   }
}

diagram attributes¶

diagram {
  // Set node metrix.
  node_width = 200;
  node_height = 100;

  // Set span metrix.
  span_width = 240;
  span_height = 120;

  // Set fontsize.
  fontsize = 20;

  A -> B [label = "Use long long\nedge label"];
  A -> C;
}

portrait mode¶

New in version 0.7.0.

{
  orientation = portrait

  A -> B -> C;
       B -> D;
}

Simple diagram for master data manager¶

diagram admin {
  index [label = "List of FOOs"];
  add [label = "Add FOO"];
  add_confirm [label = "Add FOO (confirm)"];
  edit [label = "Edit FOO"];
  edit_confirm [label = "Edit FOO (confirm)"];
  show [label = "Show FOO"];
  delete_confirm [label = "Delete FOO (confirm)"];

  index -> add  -> add_confirm  -> index;
  index -> edit -> edit_confirm -> index;
  index -> show -> index;
  index -> delete_confirm -> index;
}
